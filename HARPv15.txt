//! # HARP v15 — `foundation/storage.rs` //! //! ## v15 Changes //! - `MramNvmStorage`: replaces `QspiFlashRing` for the Merkle chain head. //! Uses two-phase transactional writes (write → verify CRC → advance pointer) //! to guarantee atomic updates under power loss. //! - `liveness_reboot_safe_posture()`: reads the last valid MRAM record and //! returns the safe joint-angle snapshot. Determine the inactive slot (not pointed to by `MRAM_HEAD_PTR_ADDR`). /// 2. Write the new record to the inactive slot. /// 3. Read back and verify CRC32 of the written data. /// 4. Advance `MRAM_HEAD_PTR_ADDR` to the new slot atomically. /// /// If power fails after step 2 but before step 4, the old valid record /// is still intact in the other slot. The next reboot will find it. pub fn commit(&mut self, state: &mut MramSafetyState) -> Result<(), ()> { let active_slot = self.read_head_ptr()?; let inactive_slot = 1 - active_slot; let addr = self.slot_addr(inactive_slot); // Compute CRC32 over all fields except the crc32 field itself. state.magic = MRAM_MAGIC; state.crc32 = self.crc32_of(state); let bytes = unsafe { core::slice::from_raw_parts( state as *const _ as *const u8, core::mem::size_of::<MramSafetyState>(), ) }; self.spi_write(addr, bytes)?; // Read-back verification. let mut readback = [0u8; core::mem::size_of::<MramSafetyState>()]; self.spi_read(addr, &mut readback)?; if readback != bytes { return Err(()); } // Advance the head pointer. self.write_head_ptr(inactive_slot)?; self.write_count += 1; Ok(()) } /// Restore the last valid safety state from MRAM. /// Called on boot to recover from power loss. pub fn load_latest(&mut self) -> Option<MramSafetyState> { for slot in 0..2u8 { let addr = self.slot_addr(slot); if state.magic == MRAM_MAGIC && state.crc32 == self.crc32_of(&state) { return Some(state); } } None } // ------------------------------------------------------------------ // Internal helpers // ------------------------------------------------------------------ fn slot_addr(&self, slot: u8) -> u32 { if slot == 0 { MRAM_SLOT_A_ADDR } else { MRAM_SLOT_B_ADDR } } fn read_head_ptr(&mut self) -> Result<u8, ()> { let mut buf = [0u8; //! # HARP v15 — `perception/conformal.rs` //! //! Split-conformal prediction engine. //! //! Provides the guarantee P(y ∈ Ĉ(x)) ≥ 1 − δ for any fresh test point, //! where δ = `CONFORMAL_DELTA`. Used by: //! - `Eskf::floor_height_bound()` — floor-height prediction intervals. //! - `TerrainNode::obstacle_bound()` — obstacle-distance prediction intervals. #![no_std] use heapless::Vec; use crate::foundation::schemas::{ConformalBound, CONFORMAL_CALIB_SIZE, CONFORMAL_DELTA}; pub struct SplitConformalEngine { /// Nonconformity scores |y_i − ŷ_i| from the calibration set. scores: Vec<f32, CONFORMAL_CALIB_SIZE>, delta: f32, /// Running mean of calibration inputs (simple linear predictor). mean: f32, count: u32, } impl SplitConformalEngine { pub fn new(delta: f32) -> Self { SplitConformalEngine { scores: Vec::new(), delta, mean: 0.0, count: 0, } } /// Ingest a calibration observation (true value `y`, predicted `y_hat`). /// Maintains the calibration set as a FIFO circular buffer. pub fn calibrate(&mut self, y: f32, y_hat: f32) { let score = (y - y_hat).abs(); if self.scores.len() == CONFORMAL_CALIB_SIZE { self.scores.remove(0); } let _ = self.scores.push(score); // Update running mean for the linear predictor. self.count += 1; self.mean += (y - self.mean) / self.count as f32; } /// Produce a prediction interval for a new input. /// /// The quantile q̂ is the ⌈(1 − δ)(n + 1)⌉ / n -th empirical quantile /// of the calibration nonconformity scores. pub fn predict(&self, y_hat: f32) -> ConformalBound { if self.scores.is_empty() { // No calibration data — return a maximally conservative interval. return ConformalBound { lo: y_hat - 999.0, hi: y_hat + 999.0, coverage: 1.0 - self.delta, }; } // Compute the (1 − δ) quantile of the score set. let n = self.scores.len() as f32; let idx = ((1.0 - self.delta) * (n + 1.0)).ceil() as usize; let idx = idx.min(self.scores.len()); // Sort copy into a fixed-size scratch buffer. let mut sorted: Vec<f32, CONFORMAL_CALIB_SIZE> = self.scores.clone(); sorted.sort_unstable_by(|a, b| a.partial_cmp(b).unwrap()); let q_hat = sorted.get(idx.saturating_sub(1)).copied().unwrap_or(f32::MAX); ConformalBound { lo: y_hat - q_hat, hi: y_hat + q_hat, coverage: 1.0 - self.delta, } } }
