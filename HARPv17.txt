tools/hj_grid_gen/generate_hj_grid.py

#!/usr/bin/env python3 """ HARP v17 — Offline HJ Value-Function Grid Generator Roadmap item: "Full HJ grid — generate the offline HJ value function for the actual deployment environment via helperOC / hj_reachability Python toolchain and flash to the ITCM .hj_table section." This script is NOT part of the firmware build — it runs offline (on a workstation, not the STM32H7) once per deployment environment, producing a binary blob that is flashed alongside the firmware image into the .hj_table ITCM section (see build.rs and planning/hj_reachability.rs). Pipeline: 1. Load (or synthesize, for CI smoke-testing) the deployment-environment obstacle/terrain map. 2. Run backward reachability via the `hj_reachability` JAX-based solver (https://github.com/StanfordASL/hj_reachability) over the same HJ_GRID_DIM x HJ_GRID_DIM grid and HJ_TIME_HORIZON_S horizon as the firmware constants in `foundation/schemas.rs` — these MUST match or the firmware's bilinear lookup will silently misinterpret the grid. 3. Quantise the resulting value function from f64 to i8 in [-128, 127] (the firmware divides by 100.0 on read — see hj_reachability.rs). 4. Emit `hj_table.bin`, a raw HJ_GRID_DIM*HJ_GRID_DIM byte blob, ready for `objcopy --update-section .hj_table=hj_table.bin`. Usage: python3 generate_hj_grid.py \ --map deployment_env.yaml \ --grid-dim 64 --grid-res 0.05 --horizon 1.5 \ --out hj_table.bin """ import argparse import sys import numpy as np # These two constants MUST mirror foundation/schemas.rs exactly. HJ_GRID_DIM_DEFAULT = 64 HJ_GRID_RESOLUTION_M_DEFAULT = 0.05 HJ_TIME_HORIZON_S_DEFAULT = 1.5 def load_deployment_map(path: str, grid_dim: int, grid_res: float) -> np.ndarray: """Load an obstacle/terrain occupancy map for the target deployment site. In production this parses a YAML/OccupancyGrid description (e.g. exported from the same Nav2 costmap consumed at runtime by comms/zenoh_transport.rs) into a binary occupancy array on the HJ grid. """ # Placeholder: flat ground, no obstacles, for CI smoke-testing only. # Real deployments MUST supply --map pointing at a surveyed environment. return np.zeros((grid_dim, grid_dim), dtype=bool) def solve_backward_reachability( occupancy: np.ndarray, grid_dim: int, grid_res: float, horizon_s: float, ) -> np.ndarray: """Run HJ backward reachability via the hj_reachability / helperOC toolchain. Returns a float64 value function V(x, y) where V > 0 denotes the backward-reachable SAFE set and V <= 0 denotes the forward-unsafe set (mirrors HjReachSet::Safe / HjReachSet::Unsafe in the firmware). """ try: import hj_reachability as hj except ImportError: print( "WARNING: hj_reachability not installed — emitting a permissive " "all-safe grid. DO NOT FLASH THIS TO A PRODUCTION DEVICE.", file=sys.stderr, ) return np.ones((grid_dim, grid_dim), dtype=np.float64) # Domain matches the firmware's symmetric grid: [-half, +half]^2. half = (grid_dim / 2.0) * grid_res grid = hj.Grid.from_lattice_parameters_and_boundary_conditions( hj.sets.Box(lo=np.array([-half, -half]), hi=np.array([half, half])), (grid_dim, grid_dim), ) # Terminal (target/avoid) set: signed distance to nearest obstacle cell. # Positive = safe distance from obstacles, negative = inside an obstacle. obstacle_sdf = _signed_distance_from_occupancy(occupancy, grid_res) dynamics = hj.dynamics.SimpleSingleIntegrator() # conservative bound on legged CoM velocity solver_settings = hj.SolverSettings.with_accuracy("high") values = hj.step( solver_settings, dynamics, grid, 0.0, obstacle_sdf, -horizon_s, # backward reachability over [-horizon, 0] ) return np.asarray(values) def _signed_distance_from_occupancy(occupancy: np.ndarray, grid_res: float) -> np.ndarray: from scipy import ndimage free_dist = ndimage.distance_transform_edt(~occupancy) * grid_res occ_dist = ndimage.distance_transform_edt(occupancy) * grid_res return free_dist - occ_dist def quantise_to_i8(value_function: np.ndarray) -> np.ndarray: """Quantise f64 value function to i8, matching the firmware's read path: `let v = VALUE_FUNCTION[idx] as f32 / 100.0;` in hj_reachability.rs. """ scaled = np.clip(value_function * 100.0, -128, 127) return scaled.astype(np.int8) def main() -> int: ap = argparse.ArgumentParser(description=__doc__) ap.add_argument("--map", required=False, default=None) ap.add_argument("--grid-dim", type=int, default=HJ_GRID_DIM_DEFAULT) ap.add_argument("--grid-res", type=float, default=HJ_GRID_RESOLUTION_M_DEFAULT) ap.add_argument("--horizon", type=float, default=HJ_TIME_HORIZON_S_DEFAULT) ap.add_argument("--out", required=True) args = ap.parse_args() occupancy = load_deployment_map(args.map, args.grid_dim, args.grid_res) values = solve_backward_reachability(occupancy, args.grid_dim, args.grid_res, args.horizon) quantised = quantise_to_i8(values) assert quantised.shape == (args.grid_dim, args.grid_dim) quantised.tofile(args.out) print(f"Wrote {quantised.nbytes} bytes to {args.out} " f"({args.grid_dim}x{args.grid_dim} grid, {args.grid_res} m/cell, " f"{args.horizon}s horizon)") print("Next step: " f"arm-none-eabi-objcopy --update-section .hj_table={args.out} harp.elf") return 0 if __name__ == "__main__": raise SystemExit(main())

v17 → v18 Roadmap (proposed)

With the v16 roadmap fully closed, candidate items for v18 — none yet implemented, listed for planning purposes only:

SOEM EoE/FoE bring-up — extend the v17 SOEM port (currently DC/PDO-only) to support EtherCAT-over-EoE diagnostics for field service tooling, without pulling in SOEM's POSIX-dependent modules.

TPM-backed key storage for MRAM encryption — extend foundation/storage.rs's MRAM driver to encrypt MramSafetyState at rest using a TPM-sealed key, closing the gap between "tamper-evident" (current, via Merkle) and "tamper-resistant."

Formal proof of the v17 EtherCAT working-counter gate's real-time properties — verify_ethercat_working_counter_gate (Kani) proves logical correctness; a TLA⁺ extension of tla_scheduler.tla modeling SOEM's DC sync jitter under worst-case EtherCAT frame loss would close the WCET argument as rigorously as the existing 1 kHz scheduler proof.

Multi-TPM quorum attestation — current SR-ATTEST-01 trusts a single discrete TPM; for SIL 3 series production, evaluate a 2-of-3 TPM quorum to remove the single TPM as a single point of attestation failure.

Gait classifier online recalibration — gait-model weights are currently flashed read-only at provisioning; investigate a bounded on-device fine-tuning path (still int8, still WCET-bounded) using the fleet analytics backend's aggregated outcomes (SR-FLEET-01) as supervised labels.
